016package com.polymathiccoder.servo.publish.influxdb.util;
017
018import java.net.InetAddress;
019import java.net.UnknownHostException;
020import java.util.Set;
021
022import com.google.common.collect.Iterables;
023import com.google.common.collect.Sets;
024import com.netflix.servo.Metric;
025import com.netflix.servo.annotations.DataSourceType;
026import com.netflix.servo.tag.BasicTag;
027import com.netflix.servo.tag.BasicTagList;
028import com.netflix.servo.tag.Tag;
029import com.polymathiccoder.servo.publish.influxdb.MetricTransformation;
030
031public final class NetflixServoUtils {
032
033    private static final Tag ATLAS_COUNTER_TAG = new BasicTag("atlas.dstype", "counter");
034    private static final Tag ATLAS_GAUGE_TAG = new BasicTag("atlas.dstype", "gauge");
035
036    private static final String CLUSTER = "nf.cluster";
037    private static final String NODE = "nf.node";
038    private static final String UNKNOWN = "unknown";
039
040    public static MetricTransformation commonTagsDecoration(){
041        return (metric) -> {
042            final Set<Tag> tags = Sets.newHashSet(Iterables.transform(metric.getConfig().getTags(),
043                    (it) -> new BasicTag(it.getKey(), it.getValue())));
044
045            // Add legacy Atlas tag
046            if (isCounter(metric)) {
047                tags.add(ATLAS_COUNTER_TAG);
048            } else if (isGauge(metric)) {
049                tags.add(ATLAS_GAUGE_TAG);
050            } else if (isRate(metric)) {
051                tags.add(ATLAS_COUNTER_TAG);
052            }
053
054            // Add cluster tag
055            final String cluster = System.getenv("NETFLIX_CLUSTER");
056            tags.add(new BasicTag(CLUSTER, (cluster == null) ? UNKNOWN : cluster));
057
058            // Add node tag
059            try {
060                tags.add(new BasicTag(NODE, InetAddress.getLocalHost().getHostName()));
061            } catch (final UnknownHostException unknownHostException) {
062                tags.add(new BasicTag(NODE, UNKNOWN));
063            }
064
065            return new Metric(metric.getConfig().getName(),
066                    BasicTagList.of(tags.toArray(new Tag[tags.size()])), metric.getTimestamp(),
067                    metric.getNumberValue());
068        };
069    }
070
071    private static boolean isCounter(final Metric metric) {
072        final String value = metric.getConfig().getTags().getValue(DataSourceType.KEY);
073        return value != null && value.equals(DataSourceType.COUNTER.name());
074    }
075
076    private static boolean isGauge(final Metric metric) {
077        final String value = metric.getConfig().getTags().getValue(DataSourceType.KEY);
078        return value != null && value.equals(DataSourceType.GAUGE.name());
079    }
080
081    private static boolean isRate(final Metric metric) {
082        final String value = metric.getConfig().getTags().getValue(DataSourceType.KEY);
083        return DataSourceType.RATE.name().equals(value)
084                || DataSourceType.NORMALIZED.name().equals(value);
085    }
086}
