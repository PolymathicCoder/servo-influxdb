016package com.polymathiccoder.servo.publish.influxdb.operations;
017
018import java.net.SocketException;
019import java.net.SocketTimeoutException;
020import java.util.List;
021
022import org.influxdb.InfluxDB.ConsistencyLevel;
023import org.influxdb.dto.BatchPoints;
024import org.influxdb.dto.Point;
025
026import com.google.gson.JsonObject;
027import com.google.gson.JsonParser;
028import com.google.gson.JsonSyntaxException;
029import com.polymathiccoder.servo.publish.influxdb.operations.error.InfluxDBClientException;
030import com.polymathiccoder.servo.publish.influxdb.operations.error.InfluxDBConnectionException;
031import com.polymathiccoder.servo.publish.influxdb.operations.error.InfluxDBOperationException;
032
033import retrofit.RetrofitError;
034import retrofit.mime.TypedByteArray;
035
036/**
037 * An Observer that forwards metrics to an InfluxDB instance.
038 *
039 * @author Abdelmonaim Remani, {@literal @}PolymathicCoder
040 *         PolymathicCoder@gmail.com
041 */
042public class InfluxDBOperations {
043
044    private final InfluxDBConnection influxDBConnection;
045
046    public InfluxDBOperations(final InfluxDBConnection timeseriesDatagateway) {
047        this.influxDBConnection = timeseriesDatagateway;
048    }
049
050    public void createDatabase() throws InfluxDBClientException {
051        final InfluxDBConfiguration configuration = influxDBConnection.getConfiguration();
052        try {
053            influxDBConnection.getDBInstance().createDatabase(configuration.getDatabaseName());
054        } catch (final RetrofitError retrofitError) {
055            handleException(
056                    retrofitError,
057                    new InfluxDBOperationException.DataDefinitionException.DatabaseCreationException(
058                            influxDBConnection.getConfiguration()));
059        }
060    }
061
062    public void writeDataInBatch(final List<Point> points) throws InfluxDBClientException {
063        final BatchPoints batchPoints = BatchPoints
064                .database(influxDBConnection.getConfiguration().getDatabaseName())
065                .retentionPolicy("default").consistency(ConsistencyLevel.ALL).build();
066
067        points.forEach((point) -> batchPoints.point(point));
068
069        try {
070            influxDBConnection.getDBInstance().write(batchPoints);
071        } catch (final RetrofitError exception) {
072            handleException(
073                    exception,
074                    new InfluxDBOperationException.DataManipulationException.DataWriteOperationException(
075                            influxDBConnection.getConfiguration(),
076                            points));
077        }
078    }
079
080    private void handleException(final RetrofitError thrown, final InfluxDBClientException toThrow) {
081        switch (thrown.getKind()) {
082        case NETWORK:
083            if (thrown.getCause() instanceof SocketException) {
084                throw new InfluxDBConnectionException.RefusedException(
085                        influxDBConnection.getConfiguration());
086            } else if (thrown.getCause() instanceof SocketTimeoutException) {
087                throw new InfluxDBConnectionException.TimedOutException(
088                        influxDBConnection.getConfiguration());
089            } else {
090                throw new InfluxDBConnectionException(
091                        influxDBConnection.getConfiguration());
092            }
093        case HTTP:
094        case CONVERSION:
095        default:
096            String reason = "";
097            if (thrown.getResponse().getBody() != null) {
098                final String responseBody = new String(((TypedByteArray) thrown.getResponse().getBody()).getBytes());
099                try {
100                    final JsonObject json = new JsonParser().parse(responseBody).getAsJsonObject();
101                    if (json.has("error")) {
102                        reason = json.get("error").getAsString();
103                    } else {
104                        reason = json.toString();
105                    }
106                } catch (final IllegalStateException | JsonSyntaxException exception) {
107                    reason = responseBody;
108                }
109            } else {
110                reason = "";
111            }
112            reason = reason.isEmpty() ? reason : " - " + reason;
113            toThrow.setReason("The service returned '" + thrown.getResponse().getStatus() + " " + thrown.getResponse().getReason() + reason + "'");
114            throw toThrow;
115        }
116    }
117}
