025import java.util.concurrent.atomic.AtomicInteger;

027import org.influxdb.dto.Point;

030

033import com.netflix.servo.annotations.DataSourceType;

048/**
049 * An Observer that forwards metrics to an InfluxDB instance.
050 *
051 * @author Abdelmonaim Remani, {@literal @}PolymathicCoder
052 *         PolymathicCoder@gmail.com
053 */
054public class InfluxDBMetricObserver implements MetricObserver {
055
056    protected static final Logger LOGGER = LoggerFactory.getLogger(InfluxDBMetricObserver.class);
057
058    private final InfluxDBOperations influxDBOperations;
059
060    private final List<MetricTransformation> metricTransformations;
061
062    private final InfluxDBMetricObserverMetrics influxDBMetricObserverMetrics;
063
064    /**
065     * This constructor creates an observer that forwards data in batches to an
066     * InfluxDB instance according to a configuration. It also allows for
067     * applying transformations to data point before it writing them to the
068     * database, such as appending additional tags. If no database exists with
069     * the name said in the configuration, it will be automatically created.
070     *
071     * @param influxDBOperations
072     *            XXXX.
073     * @param metricTransformations
074     *            A list of closures allowing for transforming metrics before
075     *            writing them to the database. The closure are applied in the
076     *            same order they were passed as parameters.
077     */
078    public InfluxDBMetricObserver(final InfluxDBOperations influxDBOperations,
079            final MetricTransformation... metricTransformations) {
080        this.influxDBOperations = influxDBOperations;
081        this.metricTransformations = metricTransformations.length != 0
082                ? Lists.newArrayList(metricTransformations) : Lists.newArrayList();
083
084        // Create the database
085        try {
086            influxDBOperations.createDatabase();
087        } catch (final InfluxDBClientException influxDBClientException) {
088            LOGGER.error(influxDBClientException.getMessage());
089        }
090
091        // Register Metrics
092        influxDBMetricObserverMetrics = new InfluxDBMetricObserverMetrics(new HashSet<>());
093    }
094
095    @Override
096    public void update(final List<Metric> metrics) {
097        final Stopwatch stopwatch = influxDBMetricObserverMetrics.influxDBReportingTimer.start();
098        influxDBMetricObserverMetrics.numMetricsTotal.addAndGet(metrics.size());
099
100        // @formatter:off
101        List<Point> points = metrics.stream()
102                .map((metric) -> metricTransformations.stream().reduce(metric,
103                        (m, mt) -> mt.apply(m), (m1, m2) -> m1))
104                .map((metric) -> Point.measurement(metric.getConfig().getName())
105                        .time(metric.getTimestamp(), TimeUnit.MILLISECONDS)
106                        .tag(metric.getConfig().getTags().asMap())
107                        .addField("value", metric.getNumberValue()).build())
108                .collect(toList());
109        // @formatter:on
110
111        try {
112            LOGGER.debug("Writing {} metrics to the InfluxDB instance", metrics.size());
113            influxDBOperations.writeDataInBatch(points);
114            influxDBMetricObserverMetrics.numMetricsSent.addAndGet(metrics.size());
115        } catch (final InfluxDBConnectionException.RefusedException
116                | InfluxDBConnectionException.TimedOutException influxDBClientException) {
117            influxDBMetricObserverMetrics.numMetricsDroppedSendTimeout.incrementAndGet();
118        } catch (final InfluxDBOperationException.DataManipulationException.DataWriteOperationException influxDBClientException) {
119            influxDBMetricObserverMetrics.numMetricsDroppedHttpErr.incrementAndGet();
120        } finally {
121            influxDBMetricObserverMetrics.numMetricsTotal.addAndGet(metrics.size());
122        }
123
124        stopwatch.stop();
125    }
126
127    @Override
128    public String getName() {
129        return "influxDBMetricObserver";
130    }
131
132    // Types
133    public static class InfluxDBMetricObserverMetrics {
134        private final Timer influxDBReportingTimer = Monitors.newTimer("influxDBReportingTime");
135
136        @Monitor(name = "numMetricsTotal", type = COUNTER)
137        private final AtomicInteger numMetricsTotal = new AtomicInteger(0);
138
139        @Monitor(name = "numMetricsSent", type = COUNTER)
140        private final AtomicInteger numMetricsSent = new AtomicInteger(0);
141
142        @Monitor(name = "numMetricsDroppedSendTimeout", type = DataSourceType.COUNTER)
143        private final AtomicInteger numMetricsDroppedSendTimeout = new AtomicInteger(0);
144
145        @Monitor(name = "numMetricsDroppedHttpErr", type = COUNTER)
146        private final AtomicInteger numMetricsDroppedHttpErr = new AtomicInteger(0);
147
148        @MonitorTags
149        private final TagList tags;
150
151        public InfluxDBMetricObserverMetrics(final Collection<Tag> tags) {
152            this.tags = SortedTagList.builder().withTags(tags).build();
153        }
154
155        {
156            Monitors.registerObject("influxDBMetricObserver", this);
157        }
158    }
159}
